ORIGIN 0
#Print string 5 times

CLR B
ADD B I5
ADD B I48
SAVE B P0
SUB B I48

LOAD C I0x300		#Set string start

:PRINTA				#Print until zero
LOAD A AC
JZ A :NEWLINE
SAVE A P0
INC C
JMP :PRINTA

:NEWLINE			#New Line
LOAD A I13
SAVE A P0
LOAD A I10
SAVE A P0

LOAD C I0x300		#Set string start

DEC B				#Check Counter and stop if 5 printed
JZ B :STOP
ADD B I48
SAVE B P0
SUB B I48
JMP :PRINTA

:STOP
HALT

MEMORY
0x300	" HELLO WORLD!"
